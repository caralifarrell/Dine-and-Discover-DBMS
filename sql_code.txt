-- DDL (Table creation and data insertion)

-- Categories table creation

CREATE TABLE Categories(
  CategoryID INT NOT NULL AUTO_INCREMENT,
  CategoryName VARCHAR(50) NOT NULL,
  PRIMARY KEY (CategoryID)
);

-- Address table creation

CREATE TABLE Address(
  AddressID INT NOT NULL AUTO_INCREMENT,
  City VARCHAR(50) NOT NULL,
  Street VARCHAR(100) NOT NULL,
  State VARCHAR(50) NOT NULL,
  Country VARCHAR(50) NOT NULL,
  ZipCode VARCHAR(20) NOT NULL,
  PRIMARY KEY (AddressID)
);

-- Restaurants table creation

CREATE TABLE Restaurants(
  RestaurantID INT NOT NULL AUTO_INCREMENT,
  PhoneNumber VARCHAR(12) NOT NULL,
  RestaurantName VARCHAR(100) NOT NULL,
  Description TEXT NOT NULL,
  Website VARCHAR(100) NOT NULL,
  BusinessOwner VARCHAR(100) NOT NULL,
  Pricing VARCHAR(50) DEFAULT NULL,
  ReviewsReceived INT DEFAULT NULL,
  FiveStarReceived INT DEFAULT NULL,
  FourStarReceived INT DEFAULT NULL,
  ThreeStarReceived INT DEFAULT NULL,
  TwoStarReceived INT DEFAULT NULL,
  OneStarReceived INT DEFAULT NULL,
  AverageRating DECIMAL(2,1) DEFAULT NULL,
  OpeningHours TIME NOT NULL,
  ClosingHours TIME NOT NULL,
  AddressID INT NOT NULL,
  CategoryID INT NOT NULL,
  PRIMARY KEY (RestaurantID),
  FOREIGN KEY (AddressID) REFERENCES Address(AddressID),
  FOREIGN KEY (CategoryID) REFERENCES Categories(CategoryID)
);

-- Users table creation

CREATE TABLE Users(
  UserID INT NOT NULL AUTO_INCREMENT,
  Username VARCHAR(50) NOT NULL,
  Password VARCHAR(50) NOT NULL,
  Email VARCHAR(50) NOT NULL,
  FirstName VARCHAR(50) DEFAULT NULL,
  LastName VARCHAR(50) DEFAULT NULL,
  ProfilePicture VARCHAR(50) DEFAULT NULL,
  RegistrationDate DATE NOT NULL,
  NumFollowers INT DEFAULT NULL,
  NumFollowing INT DEFAULT NULL,
  CityLocated VARCHAR(50) DEFAULT NULL,
  CountryLocated VARCHAR(50) DEFAULT NULL,
  PRIMARY KEY (UserID)
);

-- Reservations table creation

CREATE TABLE Reservations(
  ReservationID INT NOT NULL AUTO_INCREMENT,
  ReservedTime DATETIME NOT NULL,
  Guests INT NOT NULL,
  RestaurantID INT NOT NULL,
  UserID INT NOT NULL,
  PRIMARY KEY (ReservationID),
  FOREIGN KEY (RestaurantID) REFERENCES Restaurants(RestaurantID),
  FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- Reviews table creation

CREATE TABLE Reviews(
  ReviewID INT NOT NULL AUTO_INCREMENT,
  ReviewContent VARCHAR(255) NOT NULL,
  ReviewDate DATE NOT NULL,
  Rating INT NOT NULL,
  PhotoID VARCHAR(50) DEFAULT NULL,
  UserID INT NOT NULL,
  RestaurantID INT NOT NULL,
  PRIMARY KEY (ReviewID),
  FOREIGN KEY (UserID) REFERENCES Users(UserID),
  FOREIGN KEY (RestaurantID) REFERENCES Restaurants(RestaurantID)
);

-- Photos table creation

CREATE TABLE Photos(
  PhotoID INT NOT NULL AUTO_INCREMENT,
  PhotoAddressLink VARCHAR(255) NOT NULL,
  ReviewID INT NOT NULL,
  PRIMARY KEY (PhotoID),
  FOREIGN KEY (ReviewID) REFERENCES Reviews(ReviewID)
);

-- Comments table creation

CREATE TABLE Comments(
  CommentID INT NOT NULL AUTO_INCREMENT,
  Content TEXT NOT NULL,
  UserID INT NOT NULL,
  ReviewID INT NOT NULL,
  PRIMARY KEY (CommentID),
  FOREIGN KEY (ReviewID) REFERENCES Reviews(ReviewID),
  FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- Likes table creation

CREATE TABLE Likes(
  LikeID INT NOT NULL AUTO_INCREMENT,
  ReviewID INT NOT NULL,
  UserID INT NOT NULL,
  PRIMARY KEY (LikeID),
  FOREIGN KEY (ReviewID) REFERENCES Reviews(ReviewID),
  FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- DML (SQL Queries)

-- Query #1: The highest rated restaurant category.

SELECT C.CategoryName,
	   AVG(R.AverageRating) AS AverageRatingPerCategory
FROM Restaurants AS R
JOIN Categories AS C ON R.CategoryID = C.CategoryID
GROUP BY CategoryName
ORDER BY AverageRatingPerCategory DESC;

-- Query #2: Find the number of users registered per year.

SELECT YEAR(RegistrationDate) AS RegistrationYear,
	   COUNT(UserID) AS NumberOfRegistrations
FROM Users
GROUP BY YEAR(RegistrationDate)
ORDER BY YEAR(RegistrationDate);

-- Query #3: Find the restaurant category with the most reservations last year in 2023.

SELECT CategoryName,
	   COUNT(CategoryName) AS NumberOfTimesReserved
FROM Reservations AS Res
JOIN Restaurants AS R ON R.RestaurantID = Res.RestaurantID
JOIN Categories AS C ON C.CategoryID = R.CategoryID
WHERE YEAR(ReservedTime)= 2023
GROUP BY CategoryName
ORDER BY NumberOfTimesReserved DESC
LIMIT 1;

-- Query #4: Find the review(s) with the most number of likes for each restaurant. 

SELECT R.RestaurantID, R.RestaurantName, RV.ReviewID, RV.ReviewContent, RV.Rating, RV.UserID, MAX(L.TotalLikes) AS MaxLikes
FROM Restaurants R
JOIN Reviews RV ON R.RestaurantID = RV.RestaurantID
JOIN (
    SELECT ReviewID, COUNT(*) AS TotalLikes
    FROM Likes
    GROUP BY ReviewID
) L ON RV.ReviewID = L.ReviewID
GROUP BY R.RestaurantID, R.RestaurantName, RV.ReviewID, RV.ReviewContent, RV.Rating, RV.UserID
HAVING MAX(L.TotalLikes) = (
    SELECT MAX(L2.TotalLikes) AS MaxLikes
    FROM Restaurants R2
    JOIN Reviews RV2 ON R2.RestaurantID = RV2.RestaurantID
    JOIN (
        SELECT ReviewID, COUNT(*) AS TotalLikes
        FROM Likes
        GROUP BY ReviewID
    ) L2 ON RV2.ReviewID = L2.ReviewID
    WHERE R2.RestaurantID = R.RestaurantID
)
ORDER BY R.RestaurantID, MaxLikes DESC;

-- Query #5: Find the restaurant with the most photos posted about them.

SELECT R.RestaurantName,
	   COUNT(Rev.PhotoID) AS NumberOfPhotos
FROM Photos AS P
JOIN Reviews AS Rev ON Rev.ReviewID = P.ReviewID
JOIN Restaurants AS R ON R.RestaurantID = Rev.RestaurantID
GROUP BY R.RestaurantName
ORDER BY NumberOfPhotos DESC
LIMIT 1;

-- Query #6: Find the top 3 most popular category types based on the number of restaurants and their average ratings.

SELECT C.CategoryName AS CategoryType, 
       COUNT(R.RestaurantID) AS NumRestaurants,
       AVG(Rev.Rating) AS AverageRating
FROM Restaurants AS R
JOIN Reviews AS Rev ON R.RestaurantID = Rev.RestaurantID
JOIN Categories AS C ON R.CategoryID = C.CategoryID
GROUP BY C.CategoryName
ORDER BY NumRestaurants DESC, AverageRating DESC
LIMIT 3;

-- Query #7: The number of reservations made across all years per season. 
-- Spring runs from March 1 to May 31;
-- Summer runs from June 1 to August 31;
-- Fall (autumn) runs from September 1 to November 30; and
-- Winter runs from December 1 to February 28 (February 29 in a leap year).

SELECT
    CASE
        WHEN MONTH(ReservedTime) BETWEEN 3 AND 5 THEN 'Spring'
        WHEN MONTH(ReservedTime) BETWEEN 6 AND 8 THEN 'Summer'
        WHEN MONTH(ReservedTime) BETWEEN 9 AND 11 THEN 'Fall'
        ELSE 'Winter'
    END AS Season,
    COUNT(*) AS NumReservations
FROM
    Reservations
GROUP BY
    CASE
        WHEN MONTH(ReservedTime) BETWEEN 3 AND 5 THEN 'Spring'
        WHEN MONTH(ReservedTime) BETWEEN 6 AND 8 THEN 'Summer'
        WHEN MONTH(ReservedTime) BETWEEN 9 AND 11 THEN 'Fall'
        ELSE 'Winter'
    END
ORDER BY Season;

-- Query #11: Find the users that made as many reservations as reviews.

SELECT U.UserID, U.Username,
	COUNT(DISTINCT R.ReservationID) AS NumReservations,
	COUNT(DISTINCT RV.ReviewID) AS NumReviews
FROM Users U
LEFT JOIN Reservations R ON U.UserID = R.UserID
LEFT JOIN Reviews RV ON U.UserID = RV.UserID
GROUP BY U.UserID, U.Username
HAVING NumReservations = NumReviews AND NumReservations > 0;

-- Query #13: Find distinct countries’ average rating.

SELECT Country,
    COUNT(*) AS TotalRestaurants,
    AVG(R.AverageRating) AS Country_AverageRating

JOIN Restaurants R ON A.AddressID = R.AddressID
GROUP BY Country
ORDER BY Country_AverageRating DESC;

-- Query #14: Find the highest rated restaurant(s) within the most expensive ones in each category.

SELECT C.CategoryName, R.RestaurantID, R.RestaurantName, R.Pricing, R.AverageRating
FROM Restaurants R
JOIN Categories C ON R.CategoryID = C.CategoryID
WHERE (R.Pricing, R.CategoryID) IN (
    SELECT MAX(Pricing), CategoryID
    FROM Restaurants
    GROUP BY CategoryID
)
AND R.AverageRating = (
    SELECT MAX(AverageRating)
    FROM Restaurants
    WHERE CategoryID = R.CategoryID AND Pricing = R.Pricing
);

-- Query #15: Find the restaurants where the number of reservations exceeds the average number of reservations for all restaurants
SELECT * FROM Restaurants
WHERE RestaurantID IN (
    SELECT RestaurantID
    FROM Reservations
    GROUP BY RestaurantID
    HAVING COUNT(*) > (SELECT AVG(num_reservations) FROM (
        SELECT COUNT(*) AS num_reservations
        FROM Reservations
        GROUP BY RestaurantID
    ) a)
);

-- Query #16: Find the reviews that the restaurants received from the user(s) located in the same city.

SELECT DISTINCT Reviews.ReviewID, Restaurants.RestaurantID, Restaurants.RestaurantName, Address.AddressID,Address.City, Users.UserID, Users.UserName, Users.CityLocated

JOIN Users ON Reviews.UserID = Users.UserID

WHERE Restaurants.AddressID = Address.AddressID;

-- Query #17: Find the top 1 restaurant with the most reservations for each country.

SELECT 
    a.Country AS CountryName,
    r.RestaurantName AS TopRestaurant,
    COUNT(*) AS ReservationCount
FROM 
    Reservations res
JOIN 
    Restaurants r ON res.RestaurantID = r.RestaurantID
JOIN 
    Address a ON r.AddressID = a.AddressID
GROUP BY 
    a.Country, r.RestaurantID
HAVING 
    COUNT(*) = (
        SELECT 
            COUNT(*)
        FROM 
            Reservations res_inner
        JOIN 
            Restaurants r_inner ON res_inner.RestaurantID = r_inner.RestaurantID
        JOIN 
            Address a_inner ON r_inner.AddressID = a_inner.AddressID
        WHERE 
            a_inner.Country = a.Country
        GROUP BY 
            r_inner.RestaurantID
        ORDER BY 
            COUNT(*) DESC
        LIMIT 1
    );

-- Query #18: Find the number of reservations per restaurant.

SELECT r.RestaurantName, COUNT(*) AS ReservationCount
FROM Reservations res
JOIN Restaurants r ON res.RestaurantID = r.RestaurantID
GROUP BY r.RestaurantID
ORDER BY ReservationCount DESC;

-- Query #19: Find the best restaurant in terms of average rating and having at least a significant amount of reviews (i.e. 10 reviews).

SELECT 
    r.RestaurantName, 
    AVG(re.Rating) AS AverageRating, 
    COUNT(re.ReviewID) AS NumberOfReviews
FROM 
    Restaurants r
    JOIN Reviews re ON r.RestaurantID = re.RestaurantID
GROUP BY 
    r.RestaurantName
HAVING 
    NumberOfReviews >= 10 -- Only consider restaurants with at least 10 reviews
ORDER BY 
    AverageRating DESC, NumberOfReviews DESC
LIMIT 1;
